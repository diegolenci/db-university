------GROUP BY------

quanti iscritti ci sono stati ogni anno:
SELECT COUNT(*) AS 'numero_inscritti', YEAR('enrolment_date') AS 'anno'
FROM students
GROUP BY anno;


insegnanti che hanno ufficio nello stesso edificio:
SELECT COUNT(*) AS 'numero_insegnanti','office_address'
FROM Teachers
GROUP BY office_address;


calcolare media voti di ogni appello esame:
SELECT AVG('vote'), 'exam_id'
FROM 'exam_student'
GROUP BY 'exam_id';


quanti corsi di laurea per ogni dipartimento:
SELECT COUNT(*) AS 'numero_corsi_laurea', 'department_id'
FROM 'degrees'
GROUP BY 'department_id'


------JOIN------

studenti iscritti al corso di laurea in economia:
SELECT *
FROM students
JOIN degrees ON students.degree_id = degrees.id
WHERE degrees.name = 'Corso di Laurea in Economia';


corsi di Laurea Magistrale del Dipartimento di Neuroscienze:
SELECT *
FROM degrees
JOIN departments ON degrees.department_id = departments.id
WHERE degrees.level = 'magistrale';
AND departments.name = 'Dipartimento di Neuroscienze';


tutti i corsi in cui insegna Fulvio Amato (id=44):
SELECT *
FROM courses
JOIN course_teacher ON courses.id = course_teacher.course_id
JOIN teachers ON course_teacher.teacher_id = teachers.id
WHERE teachers.name = 'Fulvio' AND teachers.surname = 'Amato';


tutti gli studenti con i dati relativi al corso a cui si sono iscritti e il relativo dipartimento, in ordine alfabetico per nome:
SELECT *
FROM students
JOIN degrees ON students.degree_id = degrees.id
JOIN departments ON degrees.department_id = departments.id
ORDER BY students.surname, students.name;


tutti i corsi di laurea con i relativi corsi e insegnanti:
SELECT *
FROM degrees
JOIN courses ON degress.id = courses.degree_id
JOIN course_teacher ON courses.id = course_teacher.course_id
JOIN teachers ON teachers.id = course_teacher.teacher_id;


tutti i docenti che insegnano nel Dipartimento di Matematica (54):
SELECT DISTINCT teachers.name, teachers.surname
FROM teachers
JOIN course_teacher ON teachers.id = course_teacher.teacher_id
JOIN courses ON courses.id = course_teacher.course_id
JOIN degrees ON courses.degree_id = degrees.id
JOIN departments ON degrees.department_id = departments.id
WHERE departments.name = 'Dipartimento di Matematica'
